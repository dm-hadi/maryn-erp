        document.getElementById('loginForm').addEventListener('submit', async function(e) {
            e.preventDefault(); 
            const email = document.getElementById('email').value;
            const password = document.getElementById('password').value;
            // Ambil status apakah checkbox dicentang atau tidak (menghasilkan true/false)
            const isRemembered = document.getElementById('rememberMe').checked; 
            const themeStyle = getSwalTheme();

                title: 'Memproses...',

                background: themeStyle.background,

            try {
                // 1. Atur Persistence berdasarkan Checkbox Remember Me
                const persistenceType = isRemembered

                    : firebase.auth.Auth.Persistence.SESSION; // Jika tidak: Sesi hilang saat browser/tab ditutup

                await auth.setPersistence(persistenceType);
                
                // 2. Eksekusi login
                await auth.signInWithEmailAndPassword(email, password);

                let errorMessage = 'Terjadi kesalahan pada sistem.';

                    errorMessage = 'Format email tidak valid!';

                    icon: 'error',
                    title: 'Login Gagal',
                    text: errorMessage, 
                    background: themeStyle.background,
                    color: themeStyle.color,
                    confirmButtonColor: themeStyle.confirmButtonColor
                });
            }
        });

        async function shareOtomatisPilihAplikasi() {
            try {
                // Ambil Teks Versi 2 (yang disusun ke bawah tanpa spasi lebar)
                const strukContent = document.getElementById("struk-content");
                const shareText = strukContent.getAttribute("data-share-text");

                if (!shareText) {
                    showToast("Teks struk tidak ditemukan!");
                    return;
                }

                // Kirim langsung sebagai TEKS MURNI (Bukan Gambar)
                if (navigator.share) {
                    await navigator.share({
                        title: 'Struk Transaksi Maryn',
                        text: shareText
                    });
                    showToast("Membuka aplikasi cetak...");
                } else {
                    // Fallback ke WhatsApp jika browser tidak support Share
                    const encodedText = encodeURIComponent(shareText);
                    window.location.href = "https://wa.me/?text=" + encodedText;
                }
            } catch (e) {
                if(e.name !== 'AbortError') showToast("Gagal membagikan: " + e.message);
            }
        }

        function calculateTotals(filter) {
            const now = new Date();
            let sales = 0, buy = 0, totalModal = 0, totalOrders = 0;
            transactions.forEach((tx) => {
                const txDate = new Date(tx.date);
                let match = false;
                if (filter === "day") match = txDate.toDateString() === now.toDateString();
                if (filter === "month") match = txDate.getMonth() === now.getMonth() && txDate.getFullYear() === now.getFullYear();
                if (filter === "year") match = txDate.getFullYear() === now.getFullYear();
                if (match) {
                    if (tx.type === "penjualan") {
                        sales += tx.total; totalOrders++; 
                        tx.items.forEach(item => {
                            const p = products.find(x => x.name === item.name);
                            totalModal += ((p ? p.modal : 0) * item.qty);
                        });
                    } else buy += tx.total;
                }
            });
            return { sales, buy, netProfit: sales - totalModal, totalOrders };
        }

        function renderTopItems() {
            const list = document.getElementById("dash-top-items");
            if (!list) return;

            const now = new Date();
            const filtered = transactions.filter(tx => {
                if (tx.type !== "penjualan") return false;
                const txDate = new Date(tx.date);
                if (currentHomeFilter === "day") return txDate.toDateString() === now.toDateString();
                if (currentHomeFilter === "month") return txDate.getMonth() === now.getMonth() && txDate.getFullYear() === now.getFullYear();
                if (currentHomeFilter === "year") return txDate.getFullYear() === now.getFullYear();
                return true;
            });

            const itemCounts = {};
            filtered.forEach(tx => {
                tx.items.forEach(item => {
                    itemCounts[item.name] = (itemCounts[item.name] || 0) + parseInt(item.qty);
                });
            });

            const sorted = Object.entries(itemCounts)
                .sort((a, b) => b[2] - a[2])
                .slice(0, 10);

            if (sorted.length === 0) {
                list.innerHTML = '<div style="text-align:center; color:var(--text-tert); font-size:12px; padding:10px;">Belum ada data barang terjual</div>';
                return;
            }

            list.innerHTML = `<div class="leaderboard-list">` + sorted.map((item, index) => {
                const rank = index + 1;
                return `
                    <div class="top-item">
                        <div class="rank-badge rank-${rank}">${rank}</div>
                        <div class="top-info">
                            <div class="top-name">${escapeHtml(item[0])}</div>
                            <div class="top-count">${item[1]} Terjual</div>
                        </div>
                    </div>
                `;
            }).join("") + `</div>`;
        }


let promoScrollInterval;
        let promoScrollTimeout;
        let promoDirection = 1; // 1 = jalan ke kanan, -1 = jalan ke kiri

        function startPromoAutoScroll() {
            const scrollEl = document.getElementById("dash-promo-list");
            if(!scrollEl) return;

            clearInterval(promoScrollInterval);

            // Jika item sedikit dan belum melebihi lebar layar, tidak perlu animasi
            if (scrollEl.scrollWidth <= scrollEl.clientWidth) return;

            promoScrollInterval = setInterval(() => {
                // Hitung batas paling mentok sebelah kanan
                const maxScroll = scrollEl.scrollWidth - scrollEl.clientWidth;

                // Jika mentok kanan -> putar balik ke kiri
                if (scrollEl.scrollLeft >= maxScroll - 1) {
                    promoDirection = -1;
                } 
                // Jika mentok kiri -> maju lagi ke kanan
                else if (scrollEl.scrollLeft <= 0) {
                    promoDirection = 1;
                }

                // Maju atau mundur 1 pixel
                scrollEl.scrollLeft += promoDirection;
            }, 30);

            // Mencegah macet saat disentuh/ditahan oleh pengguna
            scrollEl.onmouseenter = () => clearInterval(promoScrollInterval);
            scrollEl.ontouchstart = () => {
                clearInterval(promoScrollInterval);
                clearTimeout(promoScrollTimeout);
            };
            scrollEl.onmouseleave = startPromoAutoScroll;
            scrollEl.ontouchend = () => {
                clearTimeout(promoScrollTimeout);
                // Jeda 2 detik setelah jari dilepas, baru jalan otomatis lagi
                promoScrollTimeout = setTimeout(startPromoAutoScroll, 2000);
            };
        }

        function renderPromoCarousell() {
            const list = document.getElementById("dash-promo-list");
            const section = document.getElementById("promo-section");
            if (!list || !section) return;

            const now = new Date();
            now.setHours(0,0,0,0);

            const promoProducts = products.filter(p => {
                if (!p.promoModal || !p.promoStartDate || !p.promoEndDate) return false;
                const end = new Date(p.promoEndDate); end.setHours(23, 59, 59, 999);
                return now <= end; 
            }).sort((a, b) => new Date(a.promoEndDate) - new Date(b.promoEndDate)); 

            if (promoProducts.length === 0) {
                section.style.display = "none";
                return;
            }

            section.style.display = "block";
            
            const renderCard = (p) => {
                const imgStr = p.photo ? `<img src="${p.photo}" class="promo-img">` : `<div class="promo-img" style="display:flex;align-items:center;justify-content:center;font-size:10px;color:var(--text-quat);background:var(--card-bg);">No Image</div>`;
                
                const dStart = new Date(p.promoStartDate);
                const startStr = String(dStart.getDate()).padStart(2, '0') + '/' + String(dStart.getMonth() + 1).padStart(2, '0');
                const dEnd = new Date(p.promoEndDate);
                const endStr = String(dEnd.getDate()).padStart(2, '0') + '/' + String(dEnd.getMonth() + 1).padStart(2, '0');
                
                return `
                <div class="promo-card">
                    <div class="promo-badge">PROMO</div>
                    <div class="promo-img-wrapper">
                        ${imgStr}
                    </div>
                    <div class="promo-name">${escapeHtml(p.name)}</div>
                    <div class="promo-price-old">${formatRupiah(p.modal)}</div>
                    <div class="promo-price-new">${formatRupiah(p.promoModal)}</div>
                    <div class="promo-progress">
                        <div class="promo-progress-bar"></div>
                        <span class="promo-progress-text">⏳ ${startStr} - ${endStr}</span>
                    </div>
                </div>`;
            };

            // Tampilkan normal 1x saja tanpa duplikat
            list.innerHTML = promoProducts.map(renderCard).join("");

            const allList = document.getElementById("all-promo-list");
            if(allList) {
                allList.innerHTML = promoProducts.map((p) => `<div>${renderCard(p)}</div>`).join("");
            }

            // Jalankan animasi bolak-balik
            startPromoAutoScroll();
        }

        function drawChart() {
            const svg = document.getElementById("sales-chart");
            if (!svg) return;

            // 1. Ambil lebar dan tinggi AKTUAL dari layar perangkat
            const w = svg.clientWidth || 400; 
            const h = svg.clientHeight || 160; 
            const pad = 24; 

            // 2. KUNCI UTAMA: Sinkronkan kanvas SVG dengan lebar layar asli!
            svg.setAttribute("viewBox", `0 0 ${w} ${h}`);

            // 3. Hitung jumlah hari dinamis (Tiap batang diberi ruang ~35px)
            let days = Math.floor((w - pad * 2) / 35);
            
            // Batas hari: minimal 5 hari (HP), maksimal 31 hari (PC/Layar Lebar)
            if (days < 5) days = 5;
            if (days > 31) days = 31;

            const data = [];
            for (let i = days - 1; i >= 0; i--) {
                const d = new Date(); d.setDate(d.getDate() - i);
                let sum = 0;
                transactions.forEach((tx) => { 
                    if (tx.type === "penjualan" && new Date(tx.date).toDateString() === d.toDateString()) {
                        sum += tx.total; 
                    }
                });
                data.push(sum);
            }

            const max = Math.max(...data, 1);
            
            // 4. Kalkulasi Pembagian Ruang yang Presisi
            const availableWidth = w - (pad * 2);
            const step = availableWidth / days; // Lebar porsi per 1 hari
            const barW = Math.max(step - 8, 4); // Lebar batang (dikurangi 8px untuk gap)

            let bars = "";
            data.forEach((v, i) => {
                const bh = (v / max) * (h - pad * 2.2);
                
                // Posisikan batang tepat di TENGAH-TENGAH porsinya
                const x = pad + (i * step) + (step / 2) - (barW / 2); 
                const y = h - pad - bh;
                
                const dayLabel = new Date(); 
                dayLabel.setDate(dayLabel.getDate() - (days - 1 - i));
                
                bars += '<rect x="' + x + '" y="' + y + '" width="' + barW + '" height="' + bh + '" rx="4" fill="var(--accent)" opacity="0.9"/>' +
                        '<text x="' + (x + barW / 2) + '" y="' + (h - 6) + '" text-anchor="middle" font-size="11" fill="var(--text-tert)" font-weight="800">' + dayLabel.getDate() + "</text>";
            });
            
            svg.innerHTML = bars;
        }

        function calculateSellPrice(product, platform) {
            const feeId = product[platform + "Fee"];
            if (!feeId) return 0;
            const cat = (feeCategories[platform] || []).find((f) => f.id === feeId);
            if (!cat) return 0;
            const s = settings[platform];
            if (!s) return 0;
            const totalPersen = (cat.fee + s.promo + s.premi) / 100;
            const totalNominal = s.proses + s.voucher;
            const basePrice = getActiveModal(product) * (1 + (product.markup / 100));
            return Math.round((basePrice + totalNominal) / (1 - totalPersen));
        }

        function calculateNetPrice(product, platform) {
            const sell = calculateSellPrice(product, platform);
            if (!sell) return 0;
            const cat = (feeCategories[platform] || []).find((f) => f.id === product[platform + "Fee"]);
            const s = settings[platform];
            const totalPersen = (cat.fee + s.promo + s.premi) / 100;
            return Math.round(sell - ((sell * totalPersen) + s.proses + s.voucher));
        }

        function calculateProfit(product, platform) {
            const net = calculateNetPrice(product, platform);
            const activeModal = getActiveModal(product);
            return net ? net - activeModal : 0;
        }

        function calculatePreview() {
            const rawModal = parseFloat(document.getElementById("prod-modal")?.value) || 0;
            const promoModal = parseFloat(document.getElementById("prod-promo-modal")?.value) || 0;
            const promoStart = document.getElementById("prod-promo-start")?.value;
            const promoEnd = document.getElementById("prod-promo-end")?.value;
            const markup = parseFloat(document.getElementById("prod-markup")?.value) || 0;
            
            let activeModal = rawModal;
            if (promoModal && promoStart && promoEnd) {
                const now = new Date(); now.setHours(0,0,0,0);
                const dStart = new Date(promoStart); dStart.setHours(0,0,0,0);
                const dEnd = new Date(promoEnd); dEnd.setHours(23,59,59,999);
                if (now >= dStart && now <= dEnd) activeModal = promoModal;
            }

            ["shopee", "lazada", "blibli", "tiktok"].forEach(platform => {
                const feeId = document.getElementById("prod-" + platform).value;
                const elHarga = document.getElementById("preview-" + platform + "-harga");
                const elHasil = document.getElementById("preview-" + platform + "-hasil");
                const elProfit = document.getElementById("preview-" + platform + "-profit");

                if (!elHarga || !elHasil || !elProfit) return;
                if (!activeModal || !feeId) { elHarga.textContent = elHasil.textContent = elProfit.textContent = "Rp 0"; return; }

                const cat = (feeCategories[platform] || []).find((f) => f.id === feeId);
                if (!cat) return;
                const s = settings[platform];
                const totalPersen = (cat.fee + s.promo + s.premi) / 100;
                const totalNominal = s.proses + s.voucher;
                const base = activeModal * (1 + (markup / 100));
                const sell = Math.round((base + totalNominal) / (1 - totalPersen));
                const net = Math.round(sell - ((sell * totalPersen) + totalNominal));

                elHarga.textContent = formatRupiah(sell); elHasil.textContent = formatRupiah(net); elProfit.textContent = formatRupiah(net - activeModal);
            });
        }

            };

            try {

        function updateStockFromTx(items, type) {
            items.forEach(item => {
                const p = products.find(x => x.name === item.name);
                if (p) {
                    const change = type === "penjualan" ? -item.qty : item.qty;
                    db.collection("products").doc(p.id).update({ stock: (p.stock || 0) + change }).catch(err => console.error(err));

        }

        async function saveTx() {
            if (currentTxItems.length === 0) { showToast("Tambahkan item terlebih dahulu!"); return; }
            const btnSave = document.getElementById("btn-save-tx");
            const originalText = btnSave.innerHTML;
            btnSave.disabled = true; btnSave.innerHTML = "⏳ Menyimpan...";

            const date = document.getElementById("tx-date").value || new Date().toISOString().slice(0, 10);
            const type = document.getElementById("tx-type").value;
            const platform = type === "penjualan" ? document.getElementById("tx-platform").value : ""; 
            let total = 0; currentTxItems.forEach((i) => (total += i.qty * i.price));

            const txData = { date, type, platform, items: JSON.parse(JSON.stringify(currentTxItems)), total, updatedAt: Date.now() };
            
            const isOnline = navigator.onLine; // Cek koneksi

            try {
                if (editingTxId) {
                    await db.collection("transactions").doc(editingTxId).update(txData);
                } else {
                    txData.createdAt = Date.now();
                    await db.collection("transactions").add(txData);
                    // Hanya potong stok jika ini transaksi baru, bukan saat diedit
                    updateStockFromTx(currentTxItems, type); 
                }

                closeModal("modal-tx"); 
                
                let msg = editingTxId ? "Transaksi diperbarui!" : "Transaksi disimpan!";
                if (!isOnline) msg = "Tersimpan lokal (Menunggu koneksi)"; // Timpa pesan jika offline
                
                showToast(msg);
            } catch (err) {
                showToast("Terjadi kesalahan sistem."); 
            } finally {
                btnSave.disabled = false; 
                btnSave.innerHTML = originalText; 
            }
        }

function generateStruk(tx) {
            const dailyTxs = transactions.filter(t => t.date === tx.date && t.type === tx.type).sort((a,b) => a.createdAt - b.createdAt);
            const index = dailyTxs.findIndex(t => t.id === tx.id) + 1;
            const noInvoice = (tx.type === 'penjualan' ? "INV" : "EXP") + "/" + tx.date.replace(/-/g, '') + "/" + String(index).padStart(3, '0');

            // ==================================================

                const name = item.name.substring(0, 20).padEnd(20);
                const qty = String(item.qty).padStart(3);
                const price = formatRupiah(item.price).padStart(8);
                struk += name + " " + qty + " x " + price + "\n";
            });
            
            struk += "-------------------------------------\n";
            struk += "Total".padEnd(20) + " " + formatRupiah(tx.total).padStart(16) + "\n";
            struk += "=====================================\n";
            
            if (tx.type === 'penjualan') {

                struk += "=====================================\n";

                shareText += item.name + "\n";
                shareText += item.qty + " x " + formatRupiah(item.price) + "\n";
            });
            
            shareText += "------------------------\n";
            shareText += "Total: " + formatRupiah(tx.total) + "\n";
            shareText += "------------------------\n";
            
            if (tx.type === 'penjualan') {
                shareText += "Terima Kasih Berbelanja\n";
                shareText += "Semoga menyenangkan\n";
            } else {
                shareText += "Catatan Pembelian\n";

        function calculateCalc() {
            try {
                if (!calcInput) return;
                let formula = calcInput.replace(/%/g, '/100'); 
                let result = eval(formula);
                
                if (result % 1 !== 0) {
                    result = parseFloat(result.toFixed(6)); 
                }
                
                calcInput = result.toString();
                updateCalcDisplay();
            } catch {
                calcInput = "Error";
                updateCalcDisplay();
            }

        function hitungSimulasi() {
            const ha = parseFloat(document.getElementById('sim-modal').value) || 0;
            const bp = parseFloat(document.getElementById('sim-proses').value) || 0;
            const px = (parseFloat(document.getElementById('sim-promo').value) || 0) / 100;
            const af = (parseFloat(document.getElementById('sim-selected-fee').value) || 0) / 100;
            const pr = (parseFloat(document.getElementById('sim-premi').value) || 0) / 100;
            const vc = parseFloat(document.getElementById('sim-voucher').value) || 0;
            const mu = (parseFloat(document.getElementById('sim-markup').value) || 0) / 100;

            const basePrice = ha * (1 + mu);
            const totalPersen = af + px + pr;
            const totalNominal = bp + vc;

            if (totalPersen >= 1) {
                document.getElementById('out-jual').innerText = "Fee % tinggi!";
                return;
            }

            const hargaJual = Math.round((basePrice + totalNominal) / (1 - totalPersen));
            const hasilAkhir = Math.round(hargaJual - (hargaJual * totalPersen) - totalNominal);
            const profitBersih = hasilAkhir - ha;

            document.getElementById('out-jual').innerText = "Rp " + hargaJual.toLocaleString('id-ID');
            document.getElementById('out-saldo').innerText = "Rp " + hasilAkhir.toLocaleString('id-ID');
            document.getElementById('out-profit').innerText = "Rp " + profitBersih.toLocaleString('id-ID');
        }
